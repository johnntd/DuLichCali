    function calculateCost() {

      const people = parseInt(document.getElementById("people").value) || 0;

      const days = parseInt(document.getElementById("days").value) || 0;

      const checkboxes = document.querySelectorAll('input[name="destination"]:checked');


      let totalTicket = 0;

      let totalFood = 0;

      let airportTransfer = document.getElementById("airport").checked ? 120 : 0;

      checkboxes.forEach(cb => {

        switch (cb.value) {

          case "Disneyland":

            totalTicket += 150; // base ticket, 1-day

            totalFood += 50;

            totalLodging += 80;

            break;

          case "Universal Studios":

            totalTicket += 109; // standard 1-day ticket

            totalFood += 50;

            totalLodging += 80;

            break;

          case "San Diego Zoo":

            totalTicket += 76;

            totalFood += 40;

            totalLodging += 80;

            break;

          case "Bay Area High Tech Tour":

            totalTicket += 185;

            totalFood += 60;

            totalLodging += 100;

            break;

          case "San Francisco":

            totalFood += 60;

            totalLodging += 100;

            break;

          case "Las Vegas":

            totalTicket += 90;

            totalFood += 70;

            totalLodging += 120;

            break;

          case "Grand Canyon":

            totalTicket += 110;

            totalFood += 60;

            totalLodging += 90;

            break;

          case "Yosemite":

            totalTicket += 90;

            totalFood += 50;

            totalLodging += 90;

            break;

          case "Orange County Food Tour":

            totalTicket += 30;

            totalFood += 90;

            totalLodging += 60;

            break;

          case "Bay Area Food Tour":

            totalTicket += 40;

            totalFood += 100;

            totalLodging += 70;

            break;

        }

      });


      const ticketPerPerson = totalTicket;

      const foodPerPerson = totalFood;

      const lodgingPerPerson = totalLodging;

      const dailyCostPerPerson = ticketPerPerson + foodPerPerson + lodgingPerPerson;

      const totalTripCost = people * days * dailyCostPerPerson + airportTransfer + homeCookMeal;


      document.getElementById("costBreakdown").innerHTML = `

        <p>🎟️ Vé vào cổng mỗi người/ngày: <span class="highlight">$${ticketPerPerson}</span></p>

        <p>🍜 Ăn uống mỗi người/ngày: <span class="highlight">$${foodPerPerson}</span></p>

        <p>🏨 Nhà nghỉ mỗi người/ngày: <span class="highlight">$${lodgingPerPerson}</span></p>

        <p>🚐 Phí đưa đón sân bay: <span class="highlight">$${airportTransfer}</span></p>

        <p>👩‍🍳 Bữa ăn nấu tại nhà: <span class="highlight">$${homeCookMeal}</span></p>

        <p>👉 Tổng mỗi người/ngày: <span class="highlight">$${dailyCostPerPerson}</span></p>

        <p>👥 Tổng chi phí cho ${people} người × ${days} ngày: <span class="highlight">$${totalTripCost}</span></p>

      `;
