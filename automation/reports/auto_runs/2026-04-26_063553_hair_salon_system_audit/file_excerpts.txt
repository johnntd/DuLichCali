rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // ── Helper functions ──────────────────────────────────────────────────────────
    // isVendorMember: caller is authenticated AND their vendorUsers doc maps them to this vendorId.
    // Used to restrict writes to members of the specific vendor — prevents cross-vendor tampering.
    function isVendorMember(vendorId) {
      return request.auth != null
          && exists(/databases/$(database)/documents/vendorUsers/$(request.auth.uid))
          && get(/databases/$(database)/documents/vendorUsers/$(request.auth.uid)).data.vendorId == vendorId;
    }

    // ── Vendor top-level docs ─────────────────────────────────────────────────────
    // Public read: homepage checks vendor status without requiring auth.
    // Write: only authenticated members of that specific vendor (not just any auth user).
    match /vendors/{vendorId} {
      allow read:  if true;
      allow write: if isVendorMember(vendorId);
    }

    // ── Staff schedule/availability — public read, vendor-member write ────────────
    // Customers need name/schedule/specialties to book without authenticating.
    // IMPORTANT: compensation fields (compensationType, monthlyBasePay, hireDate, payNotes)
    // are NOT stored here — they live in staffCompensation (see below).
    match /vendors/{vendorId}/staff/{doc} {
      allow read:  if true;
      allow write: if isVendorMember(vendorId);
    }

    // ── Services catalog — public read, vendor-member write ──────────────────────
    match /vendors/{vendorId}/services/{doc} {
      allow read:  if true;
      allow write: if isVendorMember(vendorId);
    }

    // ── Staff compensation — SENSITIVE, never public ──────────────────────────────
    // Contains: compensationType, monthlyBasePay, hireDate, payNotes.
    // Never publicly readable. Only authenticated members of this specific vendor.
    // Future: tighten to role='owner' only once role-check helpers are added.
    match /vendors/{vendorId}/staffCompensation/{doc} {
      allow read, write: if isVendorMember(vendorId);
    }

    // ── Payroll periods — SENSITIVE, vendor-member only ───────────────────────────
    // Contains monthly payout records: totalOwed, amountPaidOut, payrollStatus per staff.
    // Never publicly readable. Vendor-member write (owner manages payouts).
    match /vendors/{vendorId}/payrollPeriods/{doc} {
      allow read, write: if isVendorMember(vendorId);
    }

    // ── Other vendor subcollections (menuItems, notifications, etc.) ──────────────
    // Auth required. Broad auth (not vendor-membership-only) to preserve admin.html access.
    // staffCompensation is explicitly handled above with tighter rules.
    match /vendors/{vendorId}/{sub=**} {
      allow read, write: if request.auth != null;
    }

    // ── Vendor user mappings ──────────────────────────────────────────────────────
    // Auth required — no anonymous enumeration of staff emails and roles.
    // Previously: allow read: if true (needed for setup-code check) — that check
    // actually runs AFTER Firebase Auth sign-in, so auth != null is safe here.
    match /vendorUsers/{uid} {
      allow read:  if request.auth != null;
      allow write: if request.auth != null;
    }

    // ── Bookings — open (customer booking flow is unauthenticated) ────────────────
    match /bookings/{doc=**} {
      allow read, write: if true;
    }

    // Homepage availability check runs unauthenticated; drivers sign in to write.

      allow write: if request.auth != null;
